Dataset 

class MultiOutputCategorize(DisplayedTransform):
    def __init__(self, l2i=l2i, i2l=i2l):
        self.o2i = l2i
        self.i2o = i2l
        self.c = len(l2i)

    def encodes(self, o):
        return TensorCategory([self.o2i[l] for l in o])

    def decodes(self, o):
        return Category("".join([self.i2o[v.item()] for v in o]))

def label_func(x):
    return x.stem

captcha_data_block = DataBlock(
    blocks=(ImageBlock, TransformBlock(type_tfms=MultiOutputCategorize())),
    splitter=RandomSplitter(valid_pct=0.2, seed=42),
    get_y=label_func,
    item_tfms=[CropPad(size=72, pad_mode=PadMode.Border), Resize(64)],
    batch_tfms=[*aug_transforms(do_flip=False, max_rotate=5.0)],
)

dls = captcha_data_block.dataloaders(files, bs=BATCH_SIZE, num_workers=4, shuffle=True)

Model 

N_OUTPUTS = 4

class ReshapeLayer(nn.Module):
    def __init__(self, n_classes=len(l2i), n_outputs=N_OUTPUTS):
        super().__init__()
        self.n_classes = n_classes
        self.n_outputs = n_outputs
    
    def forward(self, out):
        out = out.reshape(-1, self.n_classes, self.n_outputs)

        return out

Training 

body = create_body(resnet18, cut=-2, pretrained=True)
head = create_head(body[-1][-1].bn2.num_features, N_OUTPUTS * len(l2i), ps=0.0)

model = nn.Sequential(
    body,
    head,
    ReshapeLayer(n_classes=len(l2i), n_outputs=N_OUTPUTS), # output will be [bs, num_classes=32, n_outputs=4]
)
model = model.cuda() # TODO: check if this is redundant, but for now it is better to be safe than sorry

def model_splitter(model):
    return [params(model[0]), params(model[1:])]

def multi_output_accuracy(inp, targ):
    labels_pred = torch.softmax(inp, dim=1)
    labels_pred = torch.argmax(labels_pred, dim=1)

    return ((labels_pred == targ).all(dim=1)).sum() / inp.size(0)

def multi_output_letter_accuracy(inp, targ):
    labels_pred = torch.softmax(inp, dim=1)
    labels_pred = torch.argmax(labels_pred, dim=1)

    return (labels_pred == targ).sum() / (inp.size(0) * targ.size(1))

class MultiOutputCrossEntropy(CrossEntropyLossFlat):
    def __init__(self, *args, **kwargs):
        # flatten=False turns off output flattening which is what we want
        # reduction="sum" makes the loss functions sum up the losses of individual outputs instead of averaging
        # axis=1 tells the loss function that softmax and argmax should operate on dim=1,
        # in our case this means that we go from [bs, num_classes=32, n_output=4] to [bs, n_output=4]
        # which is exactly what we want
        super().__init__(flatten=False, reduction="sum", axis=1, *args, **kwargs)
    
    def __call__(self, inp, targ, **kwargs):
        # fastai does an extra transpose inside, but we do need it, so this reverses it
        inp = inp.transpose(1, 2)
        return super().__call__(inp, targ, **kwargs)

learn = Learner(dls, model, loss_func=MultiOutputCrossEntropy(), metrics=[multi_output_accuracy, multi_output_letter_accuracy], splitter=model_splitter)
